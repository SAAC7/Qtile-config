sudo mkdir /etc/luks-keys
sudo dd bs=512 count=4 if=/dev/urandom of=/etc/luks-keys/root.key
ls /etc/luks-keys/
cat /etc/luks-keys/root.key 
sudo chmod 600 /etc/luks-keys/root.key 
lsblk
sudo cryptsetup luksAddKey /dev/sda3 /etc/luks-keys/root.key 
sudo cryptsetup luksAddKey /dev/sda3 /etc/luks-keys/root.key 
sudo vim /etc/mkinitcpio.conf
  #WE need to add the key file in the FILES part of mkinitcipio.conf 
    # FILES
    # This setting is similar to BINARIES above, however, files are added
    # as-is and are not parsed in any way.  This is useful for config files.
    FILES=(/etc/luks-keys/root.key)
sudo mkinitcpio -P
sudo vim /etc/default/grub 
  #We need to add the key file in the grub to unlock the disk
    GRUB_CMDLINE_LINUX="cryptdevice=UUID=4e8bf29c-3d34-4d48-a450-23c9f2e87d90:cryptlvm root=/dev/vgo/ROOT cryptkey=rootfs:/etc/luks-keys/root.key"
sudo grub-mkconfig -o /boot/grub/grub.cfg
sudo reboot now
